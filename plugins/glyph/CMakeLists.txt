MACRO(GLYPH name)
  ADD_LIBRARY(${name}-${TulipVersion} SHARED ${ARGN})
  TARGET_LINK_LIBRARIES(${name}-${TulipVersion} ${LibTulipCoreName} ${LibTulipOGLESName})
  INSTALL_TULIP_PLUGIN(${name}-${TulipVersion} ${TulipGlyphPluginsInstallDir})
ENDMACRO(GLYPH)
